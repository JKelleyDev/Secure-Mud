# Monsters and NPCs.
#
# Format: "[mob: <id>]" then stats.
#   name     — display name including article ("a wild boar", "the Ember Lich")
#   hp       — starting and max hit points
#   atk      — base attack power (damage = atk - target.def +/- 2)
#   defn     — defense; reduces incoming damage
#   xp       — experience awarded on kill
#   gold     — gold drop range "min-max"
#   loot     — drop table "item_id=chance, ..." (chance is 0.0-1.0)
#   respawn  — seconds before re-spawning after death (0 = never)
#   boss     — true if this is a storyline boss (gets [BOSS] tag, broadcast on kill)
#   npc      — true if this is a non-combat NPC (talkable, can't be attacked)
#
# NPCs (npc: true) ignore combat stats but still need them set;
# convention: hp: 9999, atk: 0, defn: 999, respawn: 0.
